\newcommand{\DocSectionExplain}[2]{%

    \note{#2}
  \end{frame}
}

\begin{frame}
  \frametitle{How to read this file}
  This `tests/style-matrix.tex` is a usage document for `ubc-moloch-beamer.sty`.
  It is based on the moloch beamer theme \cite{jolars-moloch}.
  \begin{itemize}
    \item 16 cases = frame type x light/dark x crest request on/off
    \item Explanations are placed in `\note{...}` only
    \item Section cases use an auto section page plus one explanation page
  \end{itemize}
  \note{
    This deck intentionally ignores thesis content and only demonstrates style behavior.
    Case keys:
    - T = title, S = section, N = normal, U = standout.
    - L = light, D = dark.
    - ON/OFF = crest requested ON/OFF.
    For title and standout frames, crest is fixed OFF by package design, even when ON is requested.
    }
  \end{frame}
\UBCSetNormalCrest{on}
